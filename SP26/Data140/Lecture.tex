\begin{align*}
	P(0) &= (1-p)^n	\\
	P(1) &= P(0) \cdot \frac{P(1)}{P(0)}\\
	P(2) &= P(1) \cdot \frac{P(2)}{P(1)} = P(1) \cdot R(2) 
.\end{align*}

This is called the consecutive odds ratios. 
\[R(k) = P(k) \cdot \frac{P(k)}{P(k-1)}, \quad k=1,2,\dots,n\] 

This is to build the probabilities from the R'S, instead of brute forcing calculations. 

In the binomial distribution: 

\[R(k) = \frac{\frac{n!}{k! (n-k)!}p^k(1-p)^{n-k}}{\frac{n!}{(k-1)!(n-k+1)!}p^{k-1}(1-p)^{n-k+1}}\]

We do this because it is a lot easier to cancel things out. 

\[R(k) = \frac{(n-k+1)p}{k(1-p)} = \left( \frac{n+1}{k} -1 \right) \frac{p}{1-p}\]

As $k$ increases, $R(k)$ decreases in $k$. 


\nt{Usually by convention we have $q = 1-p$}

this shows us that a decreasing function can only have 1 or 0 times to cross P = 1. So the binomial mode is the Largest $k$ so that $P(k) \geq P(k-1)$, because that is when it is rising. 

This happens when 

\[\left( \frac{n+1}{k} -1 \right) \frac{p}{q} \geq 1 \]
\[\frac{n+1}{k}-1 \geq \frac{1-p}{p}\]
\[\frac{n+1}{k} \ge \frac{1}{p}\]
\[k \le (n+1)p\]

Thus, the mode is the integer part of $(n+1)p$

\dfn{Method of Iterated Expectations}{
\begin{align*}
	E[T] &= \sum_{\forall T}^{}t\cdot P(T=t)\\
	     &= \sum_{\forall t}^{}t \sum_{\forall s}^{} P(S=s, T=t) \\
	     &= \sum_{\forall  t}^{} \sum_{\forall s}^{} t \cdot P(S=s)P(T=t \mid S=s) \\
	     &= \sum_{\forall s}^{} \left[\sum_{\forall t}^{} tP(t=t \mid S=s)\right] P(S=s) \\
	     &=  \sum_{\forall  s}^{} E[T \mid S=s] P(S=s)\\
	     &= E[E[T \mid S]] \\
\end{align*}
}

\dfn{Markov Property}{
Given the state today, the distribution of where you are tomorrow, depends only on your current state, irrespective of how you got there. 

\vspace{5mm}
More formally, Given the state today, the state tomorrow doesn't depend on the past. The future and the past are conditionally independent given the present. 

\[
	P(X_{n+1} = i_{n+1} \mid X_{0} = i_{0}, X_{1} = i_{1}, \dots, X_{n-1} = i_{n-1}, X_{n} = i_{n}) = P(X_{n+1} = i_{n+1} \mid X_{n} = i_{n})
\]
This is known as a \textit{one step transition probability}.

\dfn{Stationary Transition Probability}{

	\]

If you know you are at state $ i $ at time $ n $, is the same thing as if you were at state $ i $ at time $ 0 $, because the past does not matter.

\]

\ex{Application of property 4}{
