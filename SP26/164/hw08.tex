\subsection{Mixed Integer Problem}

Original Problem: 
\begin{align*}
	\min_{w \in \mathbb{R}^{d}, b \in \mathbb{R}}: \quad & \frac{1}{n} \sum_{i=1}^{n} \vert y^{(i)} - w^{T}x^{(i)} - b \vert \\
\text{s.t}: \quad & \lVert w \rVert_{0} \le k \\
\end{align*}

Reformulation:

Let $t_{i}$ be the error term $\vert y^{(i)} - w^{T}x^{(i)} - b \vert$ 

Let $z_{i}$ be a binary variable indicating the use of feature $j \in J$ 

\begin{align*}
\min_{w,b, z, t}: \quad & \frac{1}{n} \sum_{i=1}^{n} t_{i} \\
\text{s.t}: \quad &  t_{i} \ge  (y^{(i)}  - w^{T} x^{(i)} - b) \forall  i \in N \\
		  &t_{i} \ge - (y^{(i)}  - w^{T} x^{(i)} - b) \forall i \in N \\
		  & M \cdot z_j \ge w_{j} \forall j \in J \\
		  & M \cdot z_{j} \ge -w_{j} \forall j \in J\\
		  & \sum_{j=1}^{d} z_j \le k  \forall  j \in J\\
		  & z_j \in \{0,1\}
		  & t_{i} \ge 0 
\end{align*}

\[
	Q^{*}(k) = \frac{1}{L} \sum_{\ell=1}^{L} \left[ \frac{1}{\vert D_{\ell}^{\text{val}}\vert} \sum_{i \in D_{\ell}^{\text{val}}}^{} \left\vert y^{(i)} - w^{*}(k; D_{\ell}^{\text{train}})^{T}x^{(i)} - b^{*}(k; D_{\ell}^{\text{train}})\right\vert \right]
\]

Let $t_{i}$ be the residuals from the training data. 

Let $v_{i}$  be the residuals from the validation data. 

Let $z_{i}$ be a binary variable indicating the use of feature $i \in J$ 


\begin{align*}
\min_{w, b, t, z}: \quad & \frac{1}{L} \sum_{\ell = 1}^{L} \frac{1}{\vert D_{\ell}^{\text{train}}\vert} \sum_{i \in D_{\ell}^{\text{train}}}^{} t_{\ell, i}\\
\text{s.t}: \quad &  t_{\ell, i} \ge  (y^{(i)}  - w^{T}_{\ell} x^{(i)} - b_{\ell}) \forall  i \in D_{\ell}^{\text{train}} \\
		  &t_{\ell, i} \ge - (y^{(i)}  - w^{T}_{\ell} x^{(i)} - b_{\ell}) \forall i \in D_{\ell}^{\text{train}} \\
		  & Mz_{\ell, j} \ge w_{\ell, j} \forall j \in J, \ell, in L\\
		  & Mz_{\ell, j} \ge -w_{\ell, j} \forall j\in J, \ell \in  L\\
		  & \sum_{j=1}^{d} z_{\ell, j} \le k \forall \ell \\
		  & v_{\ell, i} \ge  (y^{(i)} - w^{T}_{\ell}x^{(i)} - b_{(\ell)}) \forall i \in D_{\ell}^{\text{val}} \\
		  & v_{\ell, i} \ge -(y^{(i)} - w_{\ell}^{T}x^{(i)} - b_{\ell}) \forall i \in D_{\ell}^{\text{val}}\\
		  &z_{\ell, j} \in \{0,1\}\\
		  &t_{\ell, i}, v_{\ell, i} \ge 0 \\
\end{align*}
